<!DOCTYPE html>
<html lang="ja">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>パラダイムシフトとBの操作</title>
    <style>
        .item {
            margin-bottom: 5px;
        }
        .item-value {
            display: inline-block;
            width: 30px;
            text-align: center;
        }
    </style>
</head>
<body>
    <button id="buttonA">パラダイムシフトを追加</button>
    <button id="buttonB">アーティファクト破壊</button>
    <div id="container"></div>

    <script>
        const container = document.getElementById("container");
        const buttonA = document.getElementById("buttonA");
        const buttonB = document.getElementById("buttonB");

        let items = []; // Aの配列

        function updateDisplay() {
            container.innerHTML = "";
            items.forEach((item, index) => {
                const itemDiv = document.createElement("div");
                itemDiv.className = "item";
                
                const valueSpan = document.createElement("span");
                valueSpan.className = "item-value";
                valueSpan.innerText = item.value;

                const deleteButton = document.createElement("button");
                deleteButton.innerText = "削除";
                deleteButton.onclick = () => {
                    items.splice(index, 1);
                    updateDisplay();
                };

                itemDiv.appendChild(document.createTextNode("A: "));
                itemDiv.appendChild(valueSpan);
                itemDiv.appendChild(deleteButton);
                container.appendChild(itemDiv);
            });
        }

        buttonA.onclick = () => {
            if (items.length < 9) {  // 最大9つまで
                items.push({ value: 7 });
                updateDisplay();
            } else {
                alert("パラダイムシフトの数は最大9つまでです。");
            }
        };

        buttonB.onclick = () => {
            items.forEach(item => {
                item.value = Math.max(0, item.value - 1); // 最小値は0
            });
            updateDisplay();
        };
    </script>
</body>
</html>
